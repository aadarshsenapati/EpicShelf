CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100),
  email VARCHAR(100) UNIQUE,
  phone VARCHAR(15),
  dob DATE,
  aadhaar VARCHAR(12),
  gender VARCHAR(10),
  password VARCHAR(255)
);
CREATE TABLE sellers (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100),
  email VARCHAR(100) UNIQUE,
  phone VARCHAR(15),
  dob DATE,
  aadhaar VARCHAR(12),
  gender VARCHAR(10),
  password VARCHAR(255)
);
